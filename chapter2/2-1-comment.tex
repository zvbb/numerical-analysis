\begin{zremark}
  定理1（加法的相对舍入误差定理） 证明过程中
  \begin{align*}
     & \{[S_k(1 + \rho_k) + x_{k + 1}](1 + \delta_k) - (S_k + x_{k + 1})\} / S_{k + 1} \\
     & = \delta_k + \rho_k(S_k / S_{k + 1}) (1 + \delta_k)
  \end{align*}
\end{zremark}

\textbf{证明：}
\begin{align*}
   & \{[S_k(1 + \rho_k) + x_{k + 1}](1 + \delta_k) - (S_k + x_{k + 1})\} / S_{k + 1}                                              \\
   & = \{ (S_k + S_k \rho_k + x_{k + 1}) (1 + \delta_k) - S_k - x_{k + 1} \} / S_{k + 1}                                          \\
   & = \{ S_k + S_k \rho_k + x_{k + 1} + S_k \delta_k + S_k \rho_k \delta_k + x_{k + 1} \delta_k - S_k - x_{k + 1} \} / S_{k + 1} \\
   & = \{ S_k \rho_k + S_k \delta_k + S_k \rho_k \delta_k + x_{k + 1} \delta_k \} / S_{k + 1}                                     \\
   & = \{ \delta_k(S_k + x_{k + 1}) + \rho_k S_k  ( 1 + \delta_k) \} / S_{k + 1}                                                  \\
   & = \{ \delta_k S_{k+1} + \rho_k S_k  ( 1 + \delta_k) \} / S_{k + 1}                                                           \\
   & = \delta_k + \rho_k(S_k / S_{k + 1}) (1 + \delta_k)
\end{align*}

\begin{zremark}
  定理. 设$A$为n阶方阵，则存在n阶酉阵U和V，使得
  \begin{align*}
    \|UA\|_F = \|A V\|_F = \|UAV\|_F = \|A\|_F
  \end{align*}
  ，称为$F-$范数的酉不变性。
\end{zremark}
\textbf{证明：}

我们有
\begin{align*}
  \|A\|_F^2 = \sum\limits_{i = 1}^m \sum\limits_{j = 1}^n |a_{ij}|^2 = trace(A^H A)
\end{align*}
故
\begin{align*}
  \|UA\|_F
   & = trace((UA)^H UA )  \\
   & = trace(A^H U^H U A) \\
   & = trace(A^H  A)      \\
   & = \|A\|_F
\end{align*}
同理
\begin{align*}
  \|A V\|_F
   & = trace((A V)^H (A V)) \\
   & = trace(A V (A V)^H)   \\
   & = trace(A V V^H A^H)   \\
   & = trace(A  A^H)        \\
   & = \|A\|_F
\end{align*}
最后,
\begin{align*}
  \|UAV\|_F & = \| AV\|_F = \|A\|_F
\end{align*}

\begin{zremark}
  Jordan 标准型是一个块对角矩阵，其对角元便为矩阵J 的特征值．对于特征值$\lambda_i$，
  它的代数重复度就是 Jordan 标准型中以$\lambda_i$为特征值的 Jordan 块阶数的和，而其几何重复度（即
  与$\lambda_i$相对应的线性无关的特征向量的个数）恰为以$\lambda_i$为特征值的 Jordan 块的个数．
\end{zremark}

前者易得，后者因为每个Jordan块有且仅有一个线性无关的特征向量。

考虑一个大小为$k$的Jordan块，对应的特征值为$\lambda_i$：
\begin{align*}
  J_k(\lambda_i) = \begin{pmatrix}
                     \lambda_i & 1         & 0      & \cdots & 0         \\
                     0         & \lambda_i & 1      & \cdots & 0         \\
                     \vdots    & \vdots    & \vdots & \ddots & \vdots    \\
                     0         & 0         & 0      & \cdots & \lambda_i
                   \end{pmatrix}
\end{align*}
解方程组
\begin{align*}
  (J_k(\lambda_i) - \lambda_i I) \alpha = 0
\end{align*}
我们有
\begin{align*}
  J_k(\lambda_i) - \lambda_i I
   & = \begin{pmatrix}
         0      & 1      & 0      & \cdots & 0      \\
         0      & 0      & 1      & \cdots & 0      \\
         \vdots & \vdots & \vdots & \ddots & \vdots \\
         0      & 0      & 0      & \cdots & 0
       \end{pmatrix}
\end{align*}
设
\begin{align*}
  \alpha = \begin{pmatrix}
             x_1    \\
             x_2    \\
             \vdots \\
             x_k
           \end{pmatrix}
\end{align*}
代入方程可得
\begin{equation*}
  \begin{cases}
    0 \cdot x_1 + x_2 = 0                                             \\
    0 \cdot x_1 + 0 \cdot x_2 + x_3 = 0                               \\
    \vdots                                                            \\
    0 \cdot x_1 + 0 \cdot x_2 + \cdots + 0 \cdot x_{k - 1} + x_k  = 0 \\
  \end{cases}
\end{equation*}
于是可得$x_2 = x_3 = \cdots = x_k = 0$，而$x_1$是自由变量，
故这个的Jordan块，只有1个线性无关的特征向量，形如$(1, 0, \cdots, 0)^T$。
设几何重复度为$s$，
因为Jordan标准型是对角矩阵，相当于把空间$\mathbb{C}^{n \times n}$分解为
不变子空间的直和：
\begin{align*}
  \mathbb{C}^{n \times n} = M_1 \oplus M_2 \oplus \cdots \oplus M_t
\end{align*}
其中$t$是Jordan块的总数量，对任意$\alpha \in M_i$，有$J \alpha \in M_i$.

考虑$(J - \lambda_i I) x = 0$的解空间，
如果$\lambda_j \neq \lambda_i$，则$J(\lambda_j) - \lambda_i I$是
满秩的，故只有零解，即特征向量不在$J(\lambda_j)$对应的子空间$M$中；

对每个$J(\lambda_i)$（以$\lambda_i$为n阶方阵Jordan块）只有一个非零解；

设$M'_1, M'_2, \cdots, M'_r$是$J(\lambda_i)$对应的子空间，由以上的讨论可知，
解向量$\alpha \in M'_1 \oplus M'_2 \oplus \cdots \oplus M'_r$，
显然只有当$M'_i$取各自非零解（或零）时，$J \alpha = \lambda_i \alpha$才成立，
所以解空间的维数就是$J(\lambda_i)$块的个数，即$r = s$.

综上，几何重复度（即
与$\lambda_i$相对应的线性无关的特征向量的个数）恰为以$\lambda_i$为特征值的 Jordan 块的个数，
命题得证。

\begin{zremark}
  任意$A \in \mathbb{C}^{m \times n}$，
  如果$A^H A$的秩为r，则$A^H A$的正数特征值有r个（计重数）。
\end{zremark}

\textbf{证明：}

因为$A^H A$是Hermite半正定矩阵（实数时称为实对称半正定）。
这意味着$A^H A$的特征值都是非负数，因为
对任意特征值$\lambda$，存在非零特征向量，使得，
\begin{align*}
  A^H A x     & = \lambda x      \\
  x^H A^H A x & = \lambda x^H x  \\
  (Ax)^H (Ax) & = \lambda x^H x  \\
  (Ax, Ax)    & = \lambda (x, x)
\end{align*}
因为$(x, x) > 0, (Ax, Ax) \geq 0$，于是可得
\begin{align*}
  \lambda \geq 0
\end{align*}
所以，$A^H A$的特征值满足$\lambda \geq 0$.

又因为$A^H A$是正规矩阵，可以酉对角化：
\begin{align*}
  A^H A = U D U^H, D = diag(\lambda_1, \lambda_2, \cdots, \lambda_n)
\end{align*}
于是可得：

代数重数 = D中对角线上该值的出现次数。
$A^H A$的秩（即$rank(A^H A)$）和D对角线上非零元的个数相同，

另外可知，此时零特征值的代数重数 = 零空间的维数 = $n - rank(A^H A)$.
